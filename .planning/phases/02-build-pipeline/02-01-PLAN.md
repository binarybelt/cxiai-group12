```typescript
export const TokenOverrideSchema = z.object({
  tokenId: z.string(),
  value: z.string(),
});

export const ComponentRefSchema = z.object({
  componentId: z.string(),
  props: z.record(z.string(), z.unknown()),
  tokenOverrides: z.array(TokenOverrideSchema).optional(),
});

export const SectionSchema = z.object({
  id: z.string(),
  type: z.enum(["hero", "content", "cta", "footer", "navigation", "data", "disclaimer", "isi"]),
  components: z.array(ComponentRefSchema),
  order: z.number(),
});

export const PageSpecSchema = z.object({
  id: z.string(),
  title: z.string(),
  market: z.string(),
  product: z.string(),
  sections: z.array(SectionSchema).min(1),
  metadata: PageSpecMetadataSchema,
});

export type ComponentRef = z.infer<typeof ComponentRefSchema>;
export type PageSpec = z.infer<typeof PageSpecSchema>;
```

2. Create `src/agents/brief-interpreter/schema.ts`:
   - Define `BriefInterpretationSchema` as a Zod object with fields: pageType (string, describes pattern ID), market (string, market ID), product (string), audience (enum: hcp/patient/general), contentRequirements (array of strings), toneKeywords (array of strings), mustIncludeComponents (array of strings), reasoning (string for explainability).
   - Export the schema and the inferred type `BriefInterpretation`.

3. Create `src/agents/component-selector/schema.ts`:
   - Export function `buildConstrainedPageSpecSchema()` that constructs a constrained schema at request time.
   - CRITICAL: This schema MUST derive from the existing `page-spec.ts` schemas using `.extend()` — do NOT rewrite ComponentRefSchema or PageSpecSchema from scratch. The pattern is:
     ```
     const componentIds = loadComponents().map(c => c.id) as [string, ...string[]];
     const tokenIds = loadTokens().map(t => t.id) as [string, ...string[]];
     ```
   - Constrain `componentId` via `z.enum(componentIds)` — this is the BUILD-03 enforcement.
   - Constrain `tokenId` in TokenOverrideSchema via `z.enum(tokenIds)` — this closes the token allowlisting gap (BUILD-03 completeness).
   - Add `selectionReason: z.string()` to the component ref via `.extend()` — but this field is ONLY for the generation schema. Strip it before the PageSpec reaches the renderer (the base PageSpecSchema in page-spec.ts must NOT be modified).
   - Wrap the output in `z.object({ variants: z.array(constrainedPageSpec).length(2) })` to satisfy Decision 13 (two variants).
   - Assert `componentIds.length > 0` and `tokenIds.length > 0` before constructing enums (Pitfall 2 from research).

4. Create `src/agents/component-selector/prop-shapes.ts`:
   - Export a `PROP_SHAPES` record mapping each component ID to its JSON-serializable prop contract. This tells the LLM what shapes to generate for each component.
   - For node-type props, specify the actual JSON structure:
     - ISIBlock.content: `string` (plain text safety information)
     - NavBar.links: `Array<{ label: string; href: string }>`
     - Footer.links: `Array<{ label: string; href: string }>`
     - Footer.disclaimers: `Array<string>`
     - DataTable.headers: `Array<string>`
     - DataTable.rows: `Array<Array<string>>`
     - ContentBlock.body: `string` (plain text or simple paragraphs)
   - Export `getPropShape(componentId: string)` helper that returns the prop shape for a given component.
   - This file is imported by the component-selector prompt builder to inject prop documentation into the system prompt.
