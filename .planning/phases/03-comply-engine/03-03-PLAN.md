
    2. Add imports:
       ```typescript
       import { useRef } from "react"; // add useRef to existing import
       import { ComplianceSidebar } from "@/components/compliance-sidebar";
       import type { ComplianceViolation } from "@/types/compliance";
       ```

    3. Replace the Phase 2 generation logic in handleSubmit. Instead of `submit(interp)`, use fetch:
       ```typescript
       setIsGenerating(true);
       setGateViolations(null);
       setOverrideSpec(null);

       const genRes = await fetch("/api/generate-page", {
         method: "POST",
         headers: { "Content-Type": "application/json" },
         body: JSON.stringify(interp),
       });

       if (genRes.status === 422) {
         // Compliance gate blocked — show violations
         const body = await genRes.json();
         setGateViolations(body.violations);
         // Still set variants if the spec was returned (for sidebar display)
         if (body.spec?.variants) {
           setVariants(body.spec.variants);
         }
         setPhase("error");
         setErrorMsg("Compliance gate blocked rendering. See violations in the sidebar.");
         setIsGenerating(false);
         return;
       }

       if (!genRes.ok) {
         throw new Error(`Generation failed (${genRes.status})`);
       }

       const specResult = await genRes.json();
       setVariants(specResult.variants);
       setIsGenerating(false);
       setPhase("done");
       ```

    4. Add overrideSpec state and previewRef:
       ```typescript
       const previewRef = useRef<HTMLDivElement>(null);
       const [overrideSpec, setOverrideSpec] = useState<PageSpec | null>(null);
       ```

    5. Add auto-fix handler:
       ```typescript
       function handleAutoFix(fixedSpec: PageSpec) {
         setOverrideSpec(fixedSpec);
         setGateViolations(null); // clear gate violations on fix
       }
       ```

    6. Derive currentSpec:
       ```typescript
       const currentSpec = overrideSpec ?? variants?.[selectedVariant] ?? null;
       ```

    7. Update layout from 2-column to 3-column grid on large screens:

       - On mobile: stack vertically (grid default)

    8. Wrap the PageRenderer in a div with ref:

       ```tsx
         <PageRenderer spec={currentSpec} />
       </div>
       ```

    9. Add ComplianceSidebar as the third column:
       ```tsx
       <ComplianceSidebar
         spec={currentSpec}
         previewRef={previewRef}
         onAutoFix={handleAutoFix}
       />
       ```

        ```tsx
        {phase === "error" && gateViolations && (
        ```
